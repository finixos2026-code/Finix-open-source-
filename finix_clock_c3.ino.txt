// ─────────────────────────────────────────────
//  HELPER: cek apakah waktu masuk range reminder
//  return index reminder aktif, atau -1
// ─────────────────────────────────────────────
int8_t getActiveReminder(uint8_t h, uint8_t m) {
  for (uint8_t i = 0; i < REMINDER_COUNT; i++) {
    uint16_t now   = h * 60 + m;
    uint16_t start = reminders[i].startH * 60 + reminders[i].startM;
    uint16_t end   = reminders[i].endH   * 60 + reminders[i].endM;
    if (now >= start && now < end) return (int8_t)i;
  }
  return -1;
}

// ─────────────────────────────────────────────
//  BOOT LOG
// ─────────────────────────────────────────────
void showBootLog() {
  const uint8_t MAX_VIS = 7;
  char buf[28];
  char vis[MAX_VIS][28];
  uint8_t visCnt = 0;
  memset(vis, 0, sizeof(vis));

  for (uint8_t i = 0; i < BOOT_COUNT; i++) {
    strcpy_P(buf, (char*)pgm_read_ptr(&bootLines[i]));
    buf[27] = '\0';

    if (visCnt < MAX_VIS) {
      strncpy(vis[visCnt], buf, 27);
      vis[visCnt][27] = '\0';
      visCnt++;
    } else {
      for (uint8_t j = 0; j < MAX_VIS - 1; j++) memcpy(vis[j], vis[j+1], 28);
      strncpy(vis[MAX_VIS-1], buf, 27);
      vis[MAX_VIS-1][27] = '\0';
    }

    display.clearDisplay();
    for (uint8_t r = 0; r < visCnt; r++) {
      display.setTextSize(1);
      display.setCursor(0, r * 9);
      display.print(vis[r]);
    }
    uint8_t cx = (uint8_t)(strlen(vis[visCnt-1]) * 6);
    if (cx > 120) cx = 120;
    display.fillRect(cx, (visCnt-1)*9, 5, 8, WHITE);
    display.display();

    if (buf[1]==' ' && buf[2]==' ') delay(85);
    else if (i==0 || i==BOOT_COUNT-1) delay(400);
    else delay(210);
  }
  delay(500);
}

// ─────────────────────────────────────────────
//  WIFI
// ─────────────────────────────────────────────
bool connectWiFi() {
  WiFi.mode(WIFI_STA);
  WiFi.setAutoReconnect(true);
  WiFi.begin(WIFI_SSID, WIFI_PASS);

  display.clearDisplay();
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.print("[ NET ] Connecting...");
  display.drawLine(0, 9, 127, 9, WHITE);
  display.setCursor(0, 14);
  display.printf("SSID: %s", WIFI_SSID);
  display.display();

  unsigned long t0 = millis();
  uint8_t dots = 0;
  while (WiFi.status() != WL_CONNECTED) {
    if (millis() - t0 > 12000) {
      display.setCursor(0, 38); display.print("[FAIL] No response");
      display.setCursor(0, 50); display.print("Continuing offline");
      display.display(); delay(1500);
      return false;
    }
    display.fillRect(0, 26, 128, 10, BLACK);
    display.setCursor(0, 26); display.print("Auth");
    for (uint8_t d = 0; d < (dots%12)+1; d++) display.print(".");
    dots++;
    display.display(); delay(320);
  }
  display.fillRect(0, 38, 128, 26, BLACK);
  display.setCursor(0, 38); display.print("[OK]  Connected");
  display.setCursor(0, 50); display.printf("IP: %s", WiFi.localIP().toString().c_str());
  display.display(); delay(1000);
  return true;
}

// ─────────────────────────────────────────────
//  NTP
// ─────────────────────────────────────────────
bool syncNTP() {
  configTime(GMT_OFFSET, 0, NTP_SERVER);
  display.clearDisplay();
  display.setTextSize(1);
  display.setCursor(0, 0); display.print("[ TIME ] NTP sync...");
  display.drawLine(0, 9, 127, 9, WHITE);
  display.setCursor(0, 14); display.print(NTP_SERVER);
  display.display();

  struct tm t;
  unsigned long t0 = millis();
  uint8_t dots = 0;
  while (!getLocalTime(&t)) {
    if (millis() - t0 > 10000) {
      display.setCursor(0, 26); display.print("[FAIL] NTP timeout");
      display.display(); delay(1500);
      return false;
    }
    display.fillRect(0, 26, 128, 10, BLACK);
    display.setCursor(0, 26); display.print("Wait");
    for (uint8_t d = 0; d < (dots%8)+1; d++) display.print(".");
    dots++;
    display.display(); delay(500);
  }
  char tbuf[24];
  sprintf(tbuf, "%02d:%02d:%02d WIB", t.tm_hour, t.tm_min, t.tm_sec);
  display.fillRect(0, 26, 128, 20, BLACK);
  display.setCursor(0, 26); display.print("[OK]  Synced");
  display.setCursor(0, 38); display.print(tbuf);
  display.display(); delay(1000);
  return true;
}

// ─────────────────────────────────────────────
//  SYSINFO — 2 halaman x 2.5 detik = 5 detik
// ─────────────────────────────────────────────
void showSysInfo() {
  esp_chip_info_t chip;
  esp_chip_info(&chip);

  uint32_t heapFree  = heap_caps_get_free_size(MALLOC_CAP_INTERNAL);
  uint32_t heapTotal = heap_caps_get_total_size(MALLOC_CAP_INTERNAL);
  uint8_t  ramPct    = (uint8_t)(((heapTotal-heapFree)*100UL)/heapTotal);
  uint32_t flashKB   = ESP.getFlashChipSize() / 1024;

  char ipStr[18] = "Offline";
  if (wifiOK) WiFi.localIP().toString().toCharArray(ipStr, sizeof(ipStr));

  unsigned long upMs = millis();
  uint8_t upSec = (uint8_t)((upMs/1000UL)%60);
  uint8_t upMin = (uint8_t)((upMs/60000UL)%60);

  unsigned long showUntil = millis() + 5000;
  bool          page      = false;
  unsigned long lastFlip  = millis();

  while (millis() < showUntil) {
    if (millis() - lastFlip >= 2500) { page = !page; lastFlip = millis(); }

    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 0);
    display.print(!page ? "[ HW ] Hardware Info" : "[ SW ] Software Info");
    display.drawLine(0, 9, 127, 9, WHITE);

    if (!page) {
      display.setCursor(0, 12); display.print("CHIP: ESP32-C3 RISC-V");
      display.setCursor(0, 22); display.printf("CORE:%d REV:%d %dMHz",
                                chip.cores, chip.revision, ESP.getCpuFreqMHz());
      display.setCursor(0, 32); display.printf("RAM : %luKB/%luKB",
                                heapFree/1024, heapTotal/1024);
      display.setCursor(0, 42); display.print("USE [");
      uint8_t fill = (uint8_t)((ramPct*38UL)/100);
      display.fillRect(24,43,fill,6,WHITE);
      display.drawRect(24,43,38,6,WHITE);
      display.setCursor(64,42); display.printf("]%d%%", ramPct);
      display.setCursor(0, 54); display.printf("FLSH: %luKB", flashKB);
    } else {
      display.setCursor(0, 12); display.print("OS  : FINIX v2.2");
      display.setCursor(0, 22); display.printf("WIFI: %s", wifiOK ? "ON":"OFF");
      if (wifiOK) { display.setCursor(60,22); display.printf("(%ddBm)",WiFi.RSSI()); }
      display.setCursor(0, 32); display.print("BT  : OFF (disabled)");
      display.setCursor(0, 42); display.print("IP  : "); display.print(ipStr);
      display.setCursor(0, 54); display.printf("NTP:%s UP:%02dm%02ds",
                                ntpOK?"OK":"--", upMin, upSec);
    }

    // Dot indikator
    display.fillCircle(!page?118:124, 4, 2, WHITE);
    display.drawCircle(!page?118:124, 4, 2, WHITE);
    display.fillCircle(!page?124:118, 4, 2, BLACK);
    display.drawCircle(!page?124:118, 4, 2, WHITE);

    display.display();
    delay(80);
  }
}

  char l1[24], l2[24];

  while (cnt < 3) {
    uint8_t idx = (uint8_t)random(QUOTE_PAIRS);
    bool dup = false;
    for (uint8_t i = 0; i < cnt; i++) { if (shown[i]==idx) { dup=true; break; } }

  display.setTextSize(1);

  display.drawLine(20,52,107,52,WHITE);

  display.display();
  delay(2500);
}

// ─────────────────────────────────────────────
//  DASHBOARD — UI diperbaiki
// ─────────────────────────────────────────────
void drawDashboard() {
  struct tm t;
  if (!getLocalTime(&t)) return;
  if (t.tm_sec == lastSec) return;
  lastSec = t.tm_sec;

  const char* days[] = {"SUN","MON","TUE","WED","THU","FRI","SAT"};
  char timeBuf[10], dateBuf[14];
  sprintf(timeBuf, "%02d:%02d:%02d", t.tm_hour, t.tm_min, t.tm_sec);
  sprintf(dateBuf, "%02d/%02d/%04d", t.tm_mday, t.tm_mon+1, t.tm_year+1900);

  // Cek reminder aktif
  int8_t remIdx = getActiveReminder(t.tm_hour, t.tm_min);

  // Ticker blink: detik genap vs ganjil untuk animasi reminder
  bool blink = (t.tm_sec % 2 == 0);

  display.clearDisplay();

  // ══ TOPBAR ══
  // Background topbar (invert untuk kesan bold)
  display.fillRect(0, 0, 128, 10, WHITE);
  display.setTextColor(BLACK);
  display.setTextSize(1);

  // Kiri: nama OS
  display.setCursor(2, 1);
  display.print("FINIX OS");

  // Tengah: WiFi
  display.setCursor(50, 1);
  if (wifiOK) {
    int rssi = WiFi.RSSI();
    if      (rssi > -60) display.print("W[+++]");
    else if (rssi > -75) display.print("W[++ ]");
    else                 display.print("W[+  ]");
  } else {
    display.print("W[---]");
  }

  // Kanan: BT + NTP
  display.setCursor(96, 1);
  display.print("BT:OFF");

  display.setTextColor(WHITE); // kembalikan warna teks

  // ══ AREA JAM (y: 12-38) ══
  // Jam besar rata tengah
  display.setTextSize(2);
  int16_t x1, y1; uint16_t w, h;
  display.getTextBounds(timeBuf, 0, 13, &x1, &y1, &w, &h);
  display.setCursor((OLED_W - w) / 2, 13);
  display.print(timeBuf);

  // ══ DIVIDER ══
  display.drawLine(0, 34, 127, 34, WHITE);

  // ══ AREA INFO TENGAH (y: 36-44) ══
  display.setTextSize(1);

  // Tanggal kiri
  display.setCursor(0, 36);
  display.print(dateBuf);

  // Hari kanan
  display.setCursor(98, 36);
  display.print(days[t.tm_wday]);

  // ══ DIVIDER TIPIS ══
  display.drawLine(0, 45, 127, 45, WHITE);

  // ══ STATUS TENGAH (y: 47-54) ══
  display.setCursor(0, 47);
  if (wifiOK) {
    display.printf("IP:%-15s", WiFi.localIP().toString().c_str());
  } else {
    display.print("IP: Not connected  ");
  }

  // ══ DIVIDER BAWAH ══
  display.drawLine(0, 56, 127, 56, WHITE);

  // ══ TASKBAR BAWAH (y: 57-63) — REMINDER ══
  display.setCursor(0, 57);

  if (remIdx >= 0) {
    // Ada reminder aktif — tampilkan dengan blink icon
    const Reminder& r = reminders[remIdx];
    if (blink) {
      // Frame genap: icon + teks
      display.print(r.icon);
      display.print(" ");
      display.print(r.text);
    } else {
      // Frame ganjil: invert kotak untuk efek flash
      display.fillRect(0, 56, 128, 8, WHITE);
      display.setTextColor(BLACK);
      display.setCursor(2, 57);
      display.print(r.icon);
      display.print(" ");
      display.print(r.text);
      display.setTextColor(WHITE);
    }
  } else {
    // Tidak ada reminder: tampilkan RSSI + chip label
    if (wifiOK) {
      display.printf("RSSI:%-4ddBm", WiFi.RSSI());
    } else {
      display.print("WiFi: OFFLINE  ");
    }
    display.setCursor(84, 57);
    display.print("C3-MINI");
  }

  display.display();
}

void setup() {
  Serial.begin(115200);

  // Matikan BT 100%
  btStop();
  esp_bt_controller_disable();
  esp_bt_controller_deinit();

  // CPU 80MHz — hemat daya, chip adem
  setCpuFrequencyMhz(80);

  // I2C + OLED
  Wire.begin(SDA_PIN, SCL_PIN);
  Wire.setClock(400000);

  if (!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR)) {
    while (true) delay(500);
  }
  display.setTextColor(WHITE);
  display.cp437(true);

  randomSeed(esp_random());

  // Boot sequence
  splashKali();
  splashFinix();
  showBootLog();
  wifiOK = connectWiFi();
  if (wifiOK) ntpOK = syncNTP();
  showSysInfo();
  showAuth();
  showMotivation();
}

// ─────────────────────────────────────────────
//  LOOP
// ─────────────────────────────────────────────
void loop() {
  if (wifiOK && WiFi.status() != WL_CONNECTED) WiFi.reconnect();
  drawDashboard();
  delay(180);
}
